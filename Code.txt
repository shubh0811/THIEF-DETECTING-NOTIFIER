void notifyOnThings()
{

    Serial.println("Shubh Someone in the House");
    Blynk.notify("Alert : Shubh Someone in the House");

void setup()
{

timer.setInterval(1000L,notifyOnThings); 
}
void loop()
{
  Blynk.run();
  timer.run();
}
